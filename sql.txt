Declare json varchar(max)

SELECT @json = BulkColumn
FROM OPENROWSET (BULK 'E:\Temp\Data\AllState.JSON', SINGLE_CLOB) as j

DROP PROCEDURE IF EXISTS dbo.AirplaneInsertJson
GO
CREATE PROCEDURE dbo.AirplaneInsertJson
AS BEGIN
	INSERT INTO airplanes (airplaneIataType, engineCount, enginesType, numberRegistration, planeAge, productionLine)
	SELECT airplaneIataType, engineCount, enginesType, numberRegistration, planeAge, productionLine
	FROM OPENJSON(@json)
		WITH (airplaneIataType varchar(255), engineCount int(11), enginesType varchar(255), numberRegistration varchar(255), planeAge int(11), productionLine varchar(255))
END
